```dart
RefreshIndicator(
  onRefresh: () async {
    final bloc = context.read&lt;[BlocType]&gt;();
    bloc.add(const [Event].refresh[Feature]());
    talker.info('Manual sync triggered for [feature]');
    // Wait for refresh to complete
    await bloc.stream.firstWhere((state) =&gt; state is! [Feature]Refreshing);
  },
  child: // existing list widget
)
```

2. Add LinearProgressIndicator in AppBar bottom or above list:
```dart
// In Scaffold AppBar
appBar: AppBar(
  title: const Text('[Feature]'),
  bottom: BlocBuilder&lt;[BlocType], [StateType]&gt;(
    buildWhen: (prev, curr) =&gt; prev is [Feature]Refreshing != curr is [Feature]Refreshing,
    builder: (context, state) {
      if (state is [Feature]Refreshing) {
        return const PreferredSize(
          preferredSize: Size.fromHeight(2),
          child: LinearProgressIndicator(),
        );
      }
      return const SizedBox.shrink();
    },
  ),
),
```
